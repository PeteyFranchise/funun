    const esignState = readEsignState(d.document_data ?? null)
    const docStatus = d.status

    let derivedSigners: { name: string; email: string; status: 'pending' | 'signed' | 'declined' }[] | undefined

    if (esignState?.signers && esignState.signers.length > 0) {
      // Provider-managed esign: use per-signer statuses from the esign state
      derivedSigners = esignState.signers.map(s => ({
        name: s.name,
        email: s.email,
        status: s.status,
      }))
    } else {
      // Upload-only mode: derive from document_data.contributors (split sheet)
      // or document_data.parties (hire-right AI output) if present.
      // All signers share the document-level signed status.
      const contribs = Array.isArray((d.document_data as Record<string, unknown> | null)?.contributors)
        ? ((d.document_data as Record<string, unknown>).contributors as Array<Record<string, unknown>>)

            name: String(c.name).trim(),
            email: typeof c.email === 'string' && c.email.trim() ? String(c.email).trim() : '',
            status: signerStatus,
          }))
      }
